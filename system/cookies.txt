void	Account::_displayTimestamp(void)
{
	time_t		rawtime;
	tm			*timeinfo;
	char		timeoutput[69];
	char const	*timeformat = "[%G%m%d_%H%M%S] ";

	time(&rawtime);
	timeinfo = localtime(&rawtime);
	strftime(timeoutput, 420, timeformat, timeinfo);
	std::cout << timeoutput;
}
